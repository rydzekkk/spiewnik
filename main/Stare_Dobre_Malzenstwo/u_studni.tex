\tytul{U studni}{sł. Adam Ziemianin, muz. Krzysztof Myszkowski}{Stare Dobre Małżeństwo}
\begin{text}
    Spotkamy się kiedyś u studni\\
    Wkoło będzie zielono\\
    Nasze żony będą odświętne\\
    Nawet wódkę wypić pozwolą

    Spotkamy się kiedyś u studni\\
    Takiej zwykłej - z kołowrotem\\
    Woda w niej będzie chłodna\\
    W świat uwierzymy z powrotem

    \vin Spotkamy się u studni\\
    \vin Być może że na drugim świecie\\
    \vin Bóg przecież jest łaskawy\\
    \vin I pewnie da nam tę pociechę

    Spotkamy się kiedyś u studni\\
    Z wiecznie żywą wodą\\
    Bellona też zaprosimy\\
    On przecież będzie polewał

    \vin Spotkamy się u studni\\
    \vin I będziemy znów tacy młodzi\\
    \vin Nasze żony będą piękne\\
    \vin Nam wódka nie będzie szkodzić
\end{text}
